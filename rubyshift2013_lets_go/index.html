Interfaces

A quick introduction

  package main

  import (
      "fmt"
      "math"
  )

  type Abser interface {
      Abs() float64
  }

  func main() {
      var a Abser
      f := MyFloat(-math.Sqrt2)
      v := Vertex{3, 4}

      a = f  // a MyFloat implements Abser
      a = &v // a *Vertex implements Abser
      a = v  // a Vertex, does NOT
      // implement Abser

      fmt.Println(a.Abs())
  }

  type MyFloat float64

  func (f MyFloat) Abs() float64 {
      if f < 0 {
          return float64(-f)
      }
      return float64(f)
  }

  type Vertex struct {
      X, Y float64
  }

  func (v *Vertex) Abs() float64 {
      return math.Sqrt(v.X*v.X + v.Y*v.Y)
  }

Errors

A quick introduction

  package main

  import (
      "fmt"
      "time"
  )

  type MyError struct {
      When time.Time
      What string
  }

  func (e *MyError) Error() string {
      return fmt.Sprintf("at %v, %s",
          e.When, e.What)
  }

  func run() error {
      return &MyError{
          time.Now(),
          "it didn't work",
      }
  }

  func main() {
      if err := run(); err != nil {
          fmt.Println(err)
      }
  }

Channels

A quick introduction

  package main

  import "fmt"

  func sum(a []int, c chan int) {
      sum := 0
      for _, v := range a {
          sum += v
      }
      c <- sum // send sum to c
  }

  func main() {
      a := []int{7, 2, 8, -9, 4, 0}

      c := make(chan int)
      go sum(a[:len(a)/2], c)
      go sum(a[len(a)/2:], c)
      x, y := <-c, <-c // receive from c

      fmt.Println(x, y, x+y)
  }

Select

A quick introduction

  package main

  import "fmt"

  func fibonacci(c, quit chan int) {
      x, y := 0, 1
      for {
          select {
          case c <- x:
              x, y = y, x+y
          case <-quit:
              fmt.Println("quit")
              return
          }
      }
  }

  func main() {
      c := make(chan int)
      quit := make(chan int)
      go func() {
          for i := 0; i < 10; i++ {
              fmt.Println(<-c)
          }
          quit <- 0
      }()
      fibonacci(c, quit)
  }

Good and Bad choices

Practical use of GO

Good choice:

  • Background workers
  • Websocket servers
  • SMTP, POP3 servers
  • REST API backend
  • System daemons (monitoring, etc.)

Bad choice:

  • Low level programming (drivers, etc.)
  • Rich web service
  • Complex business logic with huge amount of external libraries
  • Replace Rails, Ruby, Python, etc.
